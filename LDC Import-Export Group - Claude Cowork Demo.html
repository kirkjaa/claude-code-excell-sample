Here's your daily alert macro — paste in Alt+F11 → Insert Module: Sub JiffyReorderAlert() Dim ws As Worksheet Dim alertWs As Worksheet Dim lastRow As Long Dim i As Long, aRow As Long Set ws = ThisWorkbook.Sheets("Inventory") lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row On Error Resume Next Application.DisplayAlerts = False ThisWorkbook.Sheets("Urgent Reorders").Delete Application.DisplayAlerts = True Set alertWs = ThisWorkbook.Sheets.Add alertWs.Name = "Urgent Reorders" alertWs.Range("A1").Value = _ "Jiffy Urgent Reorders — " & _ Format(Date, "YYYY-MM-DD") alertWs.Range("A1").Font.Bold = True ws.Rows(1).Copy alertWs.Range("A2") aRow = 3 For i = 2 To lastRow If ws.Cells(i, 9).Value = "REORDER NOW" Then ws.Rows(i).Copy alertWs.Range("A" & aRow) aRow = aRow + 1 End If Next i alertWs.Columns.AutoFit alertWs.Sort.SortFields.Add _ Key:=alertWs.Range("E3"), Order:=xlAscending MsgBox aRow - 3 & " branches need reordering.", _ vbExclamation, "Jiffy Daily Alert" End Sub Run with Alt+F8 → JiffyReorderAlert each morning.
